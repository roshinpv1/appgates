public class SynchronousCommandProcessingService {
    public void processCommand(Command command) {
        int maxAttempts = 3;
        long initialDelay = 500; // milliseconds
        long backoffFactor = 2;

        for (int attempt = 0; attempt < maxAttempts; attempt++) {
            try {
                // Process the command
                processCommandInternal(command);

                // Handle exception and retry with exponential backoff

                TimeUnit.MILLISECONDS.sleep(delay);
            }
        }

            throw new CommandProcessingException("Failed to process command after " + maxAttempts + " attempts");
